#### Sprint 2.1 — Wiring Perception + Claude Vision (1.5 jour)

**Problème actuel** : Perception agent fait screenshots + OCR mais n'utilise pas Claude Vision pour comprendre l'UI.

**Solution** : Ajouter endpoint `/vision_understand` dans Perception agent

```python
# agents/perception/services/vision_service.py
import anthropic
import base64

async def understand_screen(screenshot_b64: str, goal: str) -> dict:
    """
    Claude Vision analyse le screenshot et retourne :
    - description de l'UI visible
    - éléments cliquables détectés (avec coordonnées approximatives)
    - prochain action recommandée pour atteindre le goal
    """
    client = anthropic.Anthropic()

    response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=1024,
        messages=[{
            "role": "user",
            "content": [
                {
                    "type": "image",
                    "source": {
                        "type": "base64",
                        "media_type": "image/png",
                        "data": screenshot_b64,
                    }
                },
                {
                    "type": "text",
                    "text": f"""Tu es un agent de Computer Use.
                    Goal: {goal}

                    Analyse ce screenshot et retourne JSON avec:
                    {{
                      "ui_state": "description de ce qui est visible",
                      "clickable_elements": [{{"label": "...", "x": 0, "y": 0}}],
                      "next_action": {{"type": "click|type|scroll|wait", "target": "...", "value": "..."}},
                      "goal_progress": "0-100",
                      "goal_achieved": false
                    }}"""
                }
            ]
        }]
    )
    return json.loads(response.content[0].text)
```

#### Sprint 2.2 — Computer Use Loop dans Queen (1 jour)

**Nouveau service** : `apps/queen/src/services/computer_use_loop.js`

```javascript
/**
 * La boucle principale du Computer Use :
 * 1. Screenshot
 * 2. Claude Vision analyse + recommande action
 * 3. Executor exécute l'action
 * 4. Screenshot de vérification
 * 5. Répéter jusqu'à goal_achieved ou max_iterations
 */
async function runComputerUseLoop(goal, options = {}) {
  const { maxIterations = 20, timeoutMs = 120000, workspaceId } = options;

  const session = await createAgentSession(goal, workspaceId);

  for (let i = 0; i < maxIterations; i++) {
    // 1. Screenshot
    const screenshot = await callAgent('perception', 'POST /screenshot', {});

    // Sauvegarder screenshot → R2/S3
    const screenshotUrl = await storage.upload(screenshot.data, `sessions/${session.id}/step-${i}.png`);

    // 2. Claude Vision comprend l'écran
    const vision = await callAgent('perception', 'POST /vision_understand', {
      screenshot_b64: screenshot.data,
      goal
    });

    // Broadcast temps réel → Dashboard
    eventBus.emit('computer_use.step', { sessionId: session.id, step: i, vision });

    if (vision.goal_achieved) {
      await completeSession(session.id, 'COMPLETED');
      return { success: true, steps: i + 1 };
    }

    // HITL si risque élevé (action destructive détectée)
    if (isRiskyAction(vision.next_action)) {
      const approval = await hitlManager.request({
        question: `Action risquée détectée : ${JSON.stringify(vision.next_action)}. Approuver ?`,
        options: ['Approuver', 'Refuser', 'Modifier'],
        riskScore: 0.8,
        timeout: 30000
      });
      if (approval !== 'Approuver') continue;
    }

    // 3. Executor agit
    await executeAction(vision.next_action);

    // 4. Petite pause (UI animation)
    await sleep(500);
  }

  await completeSession(session.id, 'FAILED');
  return { success: false, reason: 'max_iterations_reached' };
}
```
